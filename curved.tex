\documentclass[tikz,border=10pt]{article}
\usepackage{tikz}

\definecolor{success}{HTML}{B2F0B2}

\begin{document}
    \begin{tikzpicture}
    
    \draw[ultra thick, success] (3,0) sin (4,1);
    %% fill in the area under the curve
    \fill[success, opacity=1] (3,0) sin (4,1);%% -- (4,0) -- cycle;
    %% draw straight line 180 degrees
    \draw[ultra thick, success] (4,1) -- (8,1);
    %% inverse sin curve
    \draw[ultra thick, success] (8,1) cos (9,0);
    \fill[success, opacity=1] (8,1) cos (9,0);
    %% draw straight long line below
    \draw[ultra thick, success] (3,0) -- (9,0);
    %% fill in the area under the 2 straight lines
    \fill[success, opacity=1] (3,0) -- (9,0) -- (8,1) -- (4,1) -- cycle;
    %% draw the rectangle using "success" color
    \draw[rounded corners, draw=success] (2, 0) rectangle (15, -5) {};
    \end{tikzpicture}
\end{document}
